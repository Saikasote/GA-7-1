import pandas as pd
import numpy as np

# Load quarterly transactional data
sales = pd.read_csv('data/quarterly_sales.csv')

# Compute YoY growth
this_q = sales.query("quarter == 'Q2-2025'").revenue.sum()
last_q = sales.query("quarter == 'Q2-2024'").revenue.sum()
yoy = (this_q - last_q) / last_q

print(f"Revenue YoY: {yoy:.2%}")

\[\mathbb{E}[R_i] = R_f + \beta_i\,(\mathbb{E}[R_m] - R_f)\]
